techMiner.tables.stack(x, xlabel, ylabel=None, vlabel=None)[source]¶
>>> import pandas as pd
>>> x = pd.DataFrame({
...    'A': [0, 1, 2],
...    'B': [3, 4, 5],
...    'C': [6, 7, 8]},
...    index=['a', 'b', 'c'])
>>> x
   A  B  C
a  0  3  6
b  1  4  7
c  2  5  8

>>> stack(x, xlabel='X', ylabel='Y', vlabel='Z')
   Z  X  Y
0  0  a  A
1  3  a  B
2  6  a  C
3  1  b  A
4  4  b  B
5  7  b  C
6  2  c  A
7  5  c  B
8  8  c  C

>>> x = pd.DataFrame({
...    'A': [0, 1, 2]},
...    index=['a', 'b', 'c'])
>>> stack(x, xlabel='X')
   A  X
0  0  a
1  1  b
2  2  c

>>> stack(x.A, xlabel='X')
      A  X
   0  0  0
   1  1  1
   2  2  2
