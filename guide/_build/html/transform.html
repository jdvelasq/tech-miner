Transformation functions¶

This module contains functions that can be applied to each element of a pandas.Series object using the map function.

techMiner.transform.applyCleanupDict(x, d, sep=None)[source]¶
>>> import pandas as pd
>>> df = pd.DataFrame({
...    'f': [0, 1, 2, 3, None, 4, 5, 6, 7, 8, 9],
... })
>>> df
      f

1   1.0
2   2.0
3   3.0
4   NaN
5   4.0
6   5.0
7   6.0
8   7.0
9   8.0
10  9.0

>>> d = {'a':[0, 1, 2],
...      'b':[4, 5, 6],
...      'c':[7, 8, 9]}
>>> df.f.map(lambda x: applyCleanupDict(x, d))

techMiner.transform.makeCleanupDict(x, match=None, sep=None)[source]¶

Create a dictionary for list cleanup.

>>> import pandas as pd
>>> df = pd.DataFrame({
...    'f': ['a b c a b a',
...          'a b c a b',
...          'a b c a b',
...          'A C b',
...          'a b',
...          'a, b, c, a',
...          'a B'],
... })
>>> df
             f
0  a b c a b a
1    a b c a b
2    a b c a b
3        A C b
4          a b
5   a, b, c, a
6          a B

>>> makeCleanupDict(df.f, match=None) 
{'a b c a b': ['A C b', 'a b c a b', 'a b c a b a', 'a, b, c, a'], 'a b': ['a B', 'a b']}

>>> makeCleanupDict(df.f, match='shortest') 
{'A C b': ['A C b', 'a b c a b', 'a b c a b a', 'a, b, c, a'], 'a b': ['a B', 'a b']}

>>> makeCleanupDict(df.f, match='longest') 
{'a b c a b a': ['A C b', 'a b c a b', 'a b c a b a', 'a, b, c, a'], 'a B': ['a B', 'a b']}

>>> df = pd.DataFrame({
...    'f': ['a b, c a, b a',
...          'A b, c A, b',
...          'a b, C A, B',
...          'A C, b',
...          None,
...          'a b',
...          'a, b, c, a',
...          'a B'],
... })
>>> df
               f
0  a b, c a, b a
1    A b, c A, b
2    a b, C A, B
3         A C, b
4           None
5            a b
6     a, b, c, a
7            a B

>>> import json
>>> print(json.dumps(makeCleanupDict(df.f, match='longest', sep=','), indent=4, sort_keys=True)) 
{
   "A C": [
       "A C",
       "C A",
       "c A",
       "c a"
   ],
   "a B": [
       "A b",
       "a B",
       "a b",
       "b a"
   ],
   "b": [
       "B",
       "b"
   ]
}
